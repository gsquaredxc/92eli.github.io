var toAnimate = document.getElementById("animate-me");
var myAnimation = (function() {
    var opacitySwitch; // this var will need to be accessed privately for this function
    return function(animator) {
        var d = new Date();
        if ((d.getTime() % 1) === 0) { // run every 1 ms
            var currOpacity = parseFloat(animator.style.opacity); // convert opacity to float
            if (currOpacity >= 1) opacitySwitch = true;
            if (currOpacity <= 0) opacitySwitch = false;
            (opacitySwitch === true) ? animator.style.opacity = currOpacity - 0.01 : animator.style.opacity = currOpacity + 0.01
        }
        window.requestAnimationFrame(function() { myAnimation(animator); });
    };
})();
toAnimate.style.opacity = "1"; // have to init opacity -- default is ""
myAnimation(toAnimate);
